const express = require('express');
const cors = require('cors');
const axios = require('axios');

const app = express();
app.use(cors()); // This tells the browser "Yes, let my frontend talk to me"

// The API Route your frontend will call
app.get('/api/live-khasra-test', async (req, res) => {
    const arcgisUrl = "https://gisprod.rajasthan.gov.in/geocode/rest/services/JDA/Settlement_JDA/MapServer/0/query";

    try {
        const response = await axios.get(arcgisUrl, {
            params: {
                where: "1=1",
                outFields: "*",
                returnGeometry: true,
                outSR: 4326,
                f: "geojson",
                resultRecordCount: 10
                // NOTICE: token is REMOVED here
            },
            headers: {
                // The "Referer Trick" Headers
                'Referer': 'https://rajdharaa.rajasthan.gov.in/citizenudh/',
                'Origin': 'https://rajdharaa.rajasthan.gov.in',
                'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) Chrome/120.0.0.0'
            }
        });

        res.json(response.data);
        console.log("Referer Trick Result:", response.data.features ? "SUCCESS" : "FAILED");

    } catch (error) {
        console.error("Referer Trick failed without token:", error.message);
        res.status(500).json({ error: "Token likely required" });
    }
});
app.listen(3000, () => {
    console.log('KshetraAI Backend running on http://localhost:3000');
});
